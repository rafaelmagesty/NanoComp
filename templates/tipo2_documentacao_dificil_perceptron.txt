Faça um perceptron com DNAr. Para isso, use a documentação: https://github.com/DanielKneipp/DNAr.
Aqui está um exemplo de código que faz uma adição em DNAr:
library(DNAr)
library(ggplot2)
library(dplyr)
InputA <- 3
InputB <- 2
InputF <- 20
ID <- 0
concentracoesIniciais <- c(InputA, InputB, 0, 0)
especies <- c('A','B','C','Waste')
reacoes <- c('A -> A + C',
             'B -> B + C',
             'C -> Waste')
b <- react(
  species = especies,
  ci = concentracoesIniciais,
  reactions = reacoes,
  ki = rep(2.8e-3, 3),
  t = seq(0, 1.6e3, length.out = 50)
)
b <- b %>%
  mutate(time = time / 3600)
p <- plot_behavior(
  b,
  x_label     = 'Time (h)',
  y_label     = 'Concentration (nmol/L)',
  legend_name = 'Species',
  geom_list   = c('line', 'point'),
  species = c('A','B','C')
)
cat('Behaviour:\n')
print(b)
print(p)
